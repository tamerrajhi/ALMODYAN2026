# Phase 3 — RPC: complete_pos_sale_atomic validation + persistence
1) اقرأ:
   v_seller_profile_id := NULLIF(p_payload->>'seller_profile_id','')::uuid;
   IF v_seller_profile_id IS NULL THEN
     RAISE EXCEPTION 'seller_profile_id is required';
   END IF;

2) Validate profile:
   IF NOT EXISTS (SELECT 1 FROM profiles WHERE id=v_seller_profile_id AND is_active=true) THEN
     RAISE EXCEPTION 'Invalid seller_profile_id';
   END IF;

3) Validate seller belongs to branch (NO NEW TABLES):
   - join profiles.user_id to user_branches.user_id
   - ensure user_branches.branch_id = v_branch_id
   IF NOT EXISTS (
     SELECT 1
     FROM profiles p
     JOIN user_branches ub ON ub.user_id = p.user_id
     WHERE p.id = v_seller_profile_id
       AND ub.branch_id = v_branch_id
   ) THEN
     RAISE EXCEPTION 'Seller not assigned to branch';
   END IF;

4) Save:
   - عند INSERT/UPDATE sales: set seller_profile_id = v_seller_profile_id
   - عند INSERT/UPDATE invoices: set seller_profile_id = v_seller_profile_id
