python - <<'PY'
import pathlib, re
files = [p.strip() for p in open("/tmp/supabase_call_files.txt","r",encoding="utf-8").read().splitlines() if p.strip()]
def head(path, n=60):
  try:
    lines = pathlib.Path(path).read_text(encoding="utf-8", errors="ignore").splitlines()
  except Exception as e:
    return ["<READ_ERROR> "+str(e)]
  return lines[:n]
import_pat = re.compile(r'^\s*import\s+.*supabase.*from\s+[\'"].+[\'"]\s*;?\s*$')
destruct_pat = re.compile(r'^\s*import\s+\{\s*supabase\s*\}\s+from\s+[\'"].+[\'"]\s*;?\s*$')
any_sup_pat = re.compile(r'\bsupabase\b')

missing_import = []
has_import = []
for f in files:
  lines = head(f, 80)
  has_any_sup = any(any_sup_pat.search(x) for x in lines)
  imports = [x for x in lines if "import" in x and "supabase" in x]
  destruct = [x for x in lines if destruct_pat.search(x)]
  any_import_match = [x for x in lines if import_pat.search(x) or destruct_pat.search(x)]
  if any_import_match:
    has_import.append((f, any_import_match[:3]))
  else:
    missing_import.append((f, imports[:5]))
print("===FILES_WITH_SUPABASE_IMPORT===")
for f, imps in has_import:
  print(f)
  for x in imps: print("  " + x)
print("\n===FILES_CALL_SUPABASE_BUT_NO_IMPORT_MATCH_IN_HEAD===")
for f, imps in missing_import:
  print(f)
  if imps:
    for x in imps: print("  " + x)
PY
