C) اكتشاف sequences تلقائيًا (فلترة آمنة)
- export async function discoverResetSequences(db): Promise<string[]>
  يرجع فقط sequences في public التي اسمها ينتهي بـ '_doc_seq'
  باستخدام query:

select c.relname
from pg_class c
join pg_namespace n on n.oid=c.relnamespace
where c.relkind='S'
  and n.nspname='public'
  and c.relname like '%\_doc\_seq' escape '\'
order by c.relname;

D) بناء SQL reset (بدون تنفيذ هنا)
- export function buildFactoryResetSQL(seqs: string[]) => string
  يُنتج SQL آمن:
  BEGIN;
  TRUNCATE TABLE <كل truncateTables> RESTART IDENTITY CASCADE;
  TRUNCATE TABLE serial_counters;
  INSERT INTO serial_counters(prefix,next_seq) VALUES
    ('FSETN',1),('FSETE',1),('FSETR',1),('FSETB',1);

  -- restart discovered sequences
  لكل seq في seqs:
    ALTER SEQUENCE public."<seq>" RESTART WITH 1;

  COMMIT;

مهم: لا تعمل DROP لأي شيء. ولا تنفّذ reset فعليًا في هذا البرومبت.
